578  def processReactions(self):
579  reactions = self.sbml.getListOfReactions()
580  nr = len(reactions)
581  nx = len(self.symbols['species']['name'])
582  # stoichiometric matrix
583  self.stoichiometricMatrix = sp.SparseMatrix(sp.zeros(nx, nr))
584  self.fluxVector = sp.zeros(nr, 1)
585 
586  assignment_ids = [ass.getId()
587  for ass in self.sbml.getListOfInitialAssignments()]
588  rulevars = [rule.getVariable()
589  for rule in self.sbml.getListOfRules()
590  if rule.getFormula() != '']
591 
592  reaction_ids = [
593  reaction.getId() for reaction in reactions
594  if reaction.isSetId()
595  ]
596 
597  def getElementFromAssignment(element_id):
598  assignment = self.sbml.getInitialAssignment(
599  element_id
600  )
601  sym = sp.sympify(assignment.getFormula())
602  # this is an initial assignment so we need to use
603  # initial conditions
604  sym = sym.subs(
605  self.symbols['species']['identifier'],
606  self.symbols['species']['value']
607  )
608  return sym
609 
610  def getElementStoichiometry(element):
611  if element.isSetId():
612  if element.getId() in assignment_ids:
613  return getElementFromAssignment(element.getId())
614  elif element.getId() in rulevars:
615  return sp.sympify(element.getId())
616  else:
617  # dont put the symbol if it wont get replaced by a
618  # rule
619  return sp.sympify(element.getStoichiometry())
620  elif element.isSetStoichiometry():
621  return sp.sympify(element.getStoichiometry())
622  else:
623  return sp.sympify(1.0)
624 
625  def isConstant(specie):
626  return specie in self.constantSpecies or \
627  specie in self.boundaryConditionSpecies
628 
629  for reactionIndex, reaction in enumerate(reactions):
630  for elementList, sign in [(reaction.getListOfReactants(), -1.0),
631  (reaction.getListOfProducts(), 1.0)]:
632  elements = {}
633  for index, element in enumerate(elementList):
634  # we need the index here as we might have multiple elements
635  # for the same species
636  elements[index] = {'species': element.getSpecies()}
637  elements[index]['stoichiometry'] = getElementStoichiometry(
638  element
639  )
640 
641  for index in elements.keys():
642  if not isConstant(elements[index]['species']):
643  specieIndex = self.speciesIndex[
644  elements[index]['species']
645  ]
646  self.stoichiometricMatrix[specieIndex, reactionIndex] \
647  += sign \
648  * elements[index]['stoichiometry'] \
649  * self.speciesConversionFactor[specieIndex] \
650  / self.speciesCompartment[specieIndex]
651 
652  # usage of formulaToL3String ensures that we get "time" as time
653  # symbol
654  math = sbml.formulaToL3String(reaction.getKineticLaw().getMath())
655  try:
656  symMath = sp.sympify(math)
657  except:
658  raise SBMLException(f'Kinetic law "{math}" contains an '
659  'unsupported expression!')
660  for r in reactions:
661  elements = list(r.getListOfReactants()) \
662  + list(r.getListOfProducts())
663  for element in elements:
664  if element.isSetId() & element.isSetStoichiometry():
665  symMath = symMath.subs(
666  sp.sympify(element.getId()),
667  sp.sympify(element.getStoichiometry())
668  )
669 
670  self.fluxVector[reactionIndex] = symMath
671  if any([
672  str(symbol) in reaction_ids
673  for symbol in self.fluxVector[reactionIndex].free_symbols
674  ]):
675  raise SBMLException(
676  'Kinetic laws involving reaction ids are currently'
677  ' not supported!'
678  )

690  def processRules(self):
691  rules = self.sbml.getListOfRules()
692 
693  rulevars = getRuleVars(rules)
694  fluxvars = self.fluxVector.free_symbols
695  specvars = self.symbols['species']['identifier'].free_symbols
696  volumevars = self.compartmentVolume.free_symbols
697  compartmentvars = self.compartmentSymbols.free_symbols
698  parametervars = sp.Matrix([
699  sp.Symbol(par.getId()) for par in self.sbml.getListOfParameters()
700  ])
701  stoichvars = self.stoichiometricMatrix.free_symbols
702 
703  assignments = {}
704 
705  for rule in rules:
706  if rule.getFormula() == '':
707  continue
708  variable = sp.sympify(rule.getVariable())
709  # avoid incorrect parsing of pow(x, -1) in symengine
710  formula = sp.sympify(sbml.formulaToL3String(rule.getMath()))
711 
712  if variable in stoichvars:
713  self.stoichiometricMatrix = \
714  self.stoichiometricMatrix.subs(variable, formula)
715 
716  if variable in specvars:
717  raise SBMLException('Species assignment rules are currently'
718  ' not supported!')
719 
720  if variable in compartmentvars:
721  raise SBMLException('Compartment assignment rules are'
722  ' currently not supported!')
723 
724  if variable in parametervars:
725  try:
726  idx = self.parameterIndex[str(variable)]
727  self.symbols['parameter']['value'][idx] \
728  = float(formula)
729  except:
730  self.sbml.removeParameter(str(variable))
731  assignments[str(variable)] = formula
732 
733  if variable in fluxvars:
734  self.fluxVector = self.fluxVector.subs(variable, formula)
735 
736  if variable in volumevars:
737  self.compartmentVolume = \
738  self.compartmentVolume.subs(variable, formula)
739 
740  if variable in rulevars:
741  for nested_rule in rules:
742  nested_formula = sp.sympify(nested_rule.getFormula())
743  nested_formula = \
744  nested_formula.subs(variable, formula)
745  nested_rule.setFormula(str(nested_formula))
746 
747  for variable in assignments:
748  assignments[variable].subs(variable, formula)
749 
750  # do this at the very end to ensure we have flattened all recursive
751  # rules
752  for variable in assignments.keys():
753  self.replaceInAllExpressions(
754  sp.sympify(variable),
755  assignments[variable]
756  )
