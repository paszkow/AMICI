153  init = sp.sympify('0.0')

221 def process_pysb_expressions(model, ODE, observables, sigmas):
222  for exp in model.expressions:
223  if exp.name in observables:
224  # here we do not define a new Expression from the
225  # pysb.Expression but define an observable, so we do not need
226  # to expand observables as these can be defined as Expressions
227  y = exp
228  ODE.add_component(
229  Observable(
230  y,
231  f'{exp.name}',
232  exp.expand_expr(expand_observables=False))
233  )
234 
235  sigma_name, sigma_value = get_sigma_name_and_value(
236  model, exp.name, sigmas
237  )
238 
239  sy = sp.Symbol(sigma_name)
240  ODE.add_component(
241  SigmaY(
242  sy,
243  f'{sigma_name}',
244  sigma_value
245  )
246  )
247 
248  my = sp.Symbol(f'm{exp.name}')
249  pi = sp.sympify('pi')
250  ODE.add_component(
251  LogLikelihood(
252  sp.Symbol(f'llh_{exp.name}'),
253  f'llh_{exp.name}',
254  0.5*sp.log(2*pi*sy**2) + 0.5*((y - my)/sy)**2
255  )
256  )
257 
258  elif exp.name in sigmas.values():
259  # do nothing
260  pass
261  else:
262  # here we do define a new Expression from the pysb.Expression
263  # so we do need to expand observables as these would otherwise
264  # lead to dependencies between different Expressions
265  ODE.add_component(
266  Expression(
267  exp,
268  f'{exp.name}',
269  exp.expand_expr(expand_observables=True))
270  )

294 def get_sigma_name_and_value(model, name, sigmas):
295  if name in sigmas:
296  if sigmas[name] not in model.expressions:
297  raise Exception(f'value of sigma {name} is not a '
298  f'valid expression.')
299  sigma_name = model.expressions[sigmas[name]].name
300  sigma_value = model.expressions[sigmas[name]].expand_expr()
301  else:
302  sigma_name = f'sigma_{name}'
303  sigma_value = sp.sympify(1.0)
304 
305  return sigma_name, sigma_value
