27 void Solver::setup(AmiVector *x, AmiVector *dx, AmiVectorArray *sx, AmiVectorArray *sdx, Model *model) {
28 
29  model->initialize(x, dx);
30 
31  /* Create solver memory object */
32  allocateSolver();
33  if (!solverMemory)
34  throw AmiException("Failed to allocated solver memory!");
35 
36  /* Initialize CVodes/IDAs solver*/
37  init(x, dx, model->t0());
38 
39  applyTolerances();
40 
41  /* Set optional inputs */
42  setErrHandlerFn();
43  /* attaches userdata*/
44  setUserData(model);
45  /* specify maximal number of steps */
46  setMaxNumSteps(maxsteps);
47  /* activates stability limit detection */
48  setStabLimDet(stldet);
49 
50  rootInit(model->ne);
51 
52  initializeLinearSolver(model);
53 
54  if (sensi >= SensitivityOrder::first && model->nx > 0) {
55  /* initialise sensitivities, this can either be user provided or
56  * come from the model definition */
57  auto sx0 = model->getInitialStateSensitivities();
58  if (sx0.empty()) {
59  model->fsx0(sx, x);
60  } else {
61  for (int ip = 0; ip < model->nplist(); ip++) {
62  for (int ix = 0; ix < model->nx; ix++) {
63  sx->at(ix,ip) =
64  (realtype)sx0.at(ix + model->nx * ip);
65  }
66  }
67  }
68 
69  model->fsdx0();
70 
71  auto plist = model->getParameterList();
72 
73  if (sensi_meth == SensitivityMethod::forward && !plist.empty()) {
74  /* Set sensitivity analysis optional inputs */
75  auto par = model->getUnscaledParameters();
76 
77  /* Activate sensitivity calculations */
78  sensInit1(sx, sdx, plist.size());
79  setSensParams(par.data(), nullptr, plist.data());
80 
81  applyTolerancesFSA();
82  } else if (sensi_meth == SensitivityMethod::adjoint) {
83  /* Allocate space for the adjoint computation */
84  adjInit();
85  }
86  }
87 
88  setId(model);
89  setSuppressAlg(true);
90  /* calculate consistent DAE initial conditions (no effect for ODE) */
91  if(model->nt()>1)
92  calcIC(model->t(1), x, dx);
93 }

100 void Solver::setupAMIB(BackwardProblem *bwd, Model *model) {
101  if (!solverMemory)
102  throw AmiException("Solver for the forward problem must be setup first");
103 
104  /* write initial conditions */
105  std::vector<realtype> dJydx = bwd->getdJydx();
106  AmiVector *xB = bwd->getxBptr();
107  xB->reset();
108  for (int ix = 0; ix < model->nxtrue; ++ix)
109  for (int iJ = 0; iJ < model->nJ; ++iJ)
110  xB->at(ix + iJ * model->nxtrue) +=
111  dJydx.at(iJ + ( ix + (model->nt() - 1) * model->nx ) * model->nJ);
112  bwd->getdxBptr()->reset();
113  bwd->getxQBptr()->reset();
114 
115  /* allocate memory for the backward problem */
116  allocateSolverB(bwd->getwhichptr());
117 
118  /* initialise states */
119  binit(bwd->getwhich(), bwd->getxBptr(), bwd->getdxBptr(), bwd->gett());
120 
121  /* Attach user data */
122  setUserDataB(bwd->getwhich(), model);
123 
124  /* Number of maximal internal steps */
125  setMaxNumStepsB(bwd->getwhich(), (maxstepsB == 0) ? maxsteps * 100 : maxstepsB);
126 
127  initializeLinearSolverB(model, bwd->getwhich());
128 
129  /* Initialise quadrature calculation */
130  qbinit(bwd->getwhich(), bwd->getxQBptr());
131 
132  applyTolerancesASA(bwd->getwhich());
133  applyQuadTolerancesASA(bwd->getwhich());
134 
135  setStabLimDetB(bwd->getwhich(), stldet);
136 }

149 void Solver::wrapErrHandlerFn(int error_code, const char *module,
150  const char *function, char *msg, void * /*eh_data*/) {
151  char buffer[250];
152  char buffid[250];
153  sprintf(buffer, "AMICI ERROR: in module %s in function %s : %s ", module,
154  function, msg);
155  switch (error_code) {
156  case 99:
157  sprintf(buffid, "AMICI:mex:%s:%s:WARNING", module, function);
158  break;
159 
160  case -1:
161  sprintf(buffid, "AMICI:mex:%s:%s:TOO_MUCH_WORK", module, function);
162  break;
163 
164  case -2:
165  sprintf(buffid, "AMICI:mex:%s:%s:TOO_MUCH_ACC", module, function);
166  break;
167 
168  case -3:
169  sprintf(buffid, "AMICI:mex:%s:%s:ERR_FAILURE", module, function);
170  break;
171 
172  case -4:
173  sprintf(buffid, "AMICI:mex:%s:%s:CONV_FAILURE", module, function);
174  break;
175 
176  default:
177  sprintf(buffid, "AMICI:mex:%s:%s:OTHER", module, function);
178  break;
179  }
180 
181  warnMsgIdAndTxt(buffid, buffer);
182 }

257

259  if (status != AMICI_SUCCESS) return;

261

440 void Solver::applyQuadTolerancesASA(int which) {
